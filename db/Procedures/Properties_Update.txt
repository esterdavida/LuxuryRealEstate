CREATE PROCEDURE [dbo].[Properties_Update]
    @Id INT,                       -- חובה לקבל את ה-ID כדי לדעת איזה נכס לעדכן
    @Title NVARCHAR(100),
    @Description NVARCHAR(MAX),
    @Price DECIMAL(18,2),
    @City NVARCHAR(50),
    @StatusId INT,
    @CategoryId INT
AS
BEGIN
    UPDATE Properties
    SET 
        Title = @Title,
        Description = @Description,
        Price = @Price,
        City = @City,
        StatusId = @StatusId,
        CategoryId = @CategoryId
    WHERE Id = @Id;              
END
GO
