CREATE PROCEDURE [dbo].[Properties_Create]
    @Title NVARCHAR(100),
    @Description NVARCHAR(MAX),
    @Price DECIMAL(18,2),
    @City NVARCHAR(50),
    @StatusId INT,
    @CategoryId INT
AS
BEGIN
    INSERT INTO Properties (Title, Description, Price, City, StatusId, CategoryId)
    VALUES (@Title, @Description, @Price, @City, @StatusId, @CategoryId);
END
GO
